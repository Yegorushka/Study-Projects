Функция СозданиеМассива(КоличествоЭлементов, МаксимЧислоМассива)
	
	М = Новый Массив;
	Генератор = Новый ГенераторСлучайныхЧисел;
	
	Для Индекс = 0 По КоличествоЭлементов Цикл	
		СлучЧисло = Генератор.СлучайноеЧисло(0, МаксимЧислоМассива);
		М.Добавить(СлучЧисло);	
	КонецЦикла;	
	
	Возврат М

	КонецЦикла;

Функция СреднееЗначениеМассива(Массив)

		Возврат СуммаЭлементовМассива(Массив) / Массив.Количество();

		Возврат Неопределено;
	КонецЕсли;

	Максимум = Массив[0];

	КонецЦикла;
